top_sources <- df %>%
  group_by(`Source / Medium`) %>%
  summarise(TotalRevenue = sum(Revenue, na.rm = TRUE)) %>%

top3_revenue <- df %>%

2.Loop over each column and compute mean, median, sd, variance

for (col_name in names(diabetes)) {
  cat(paste0("---- Statistics for '", col_name, "' ----\n"))
  
  # only compute on numeric columns
  if (is.numeric(diabetes[[col_name]])) {
    stats_vector <- c(
      Mean     = mean(diabetes[[col_name]], na.rm = TRUE),
      Median   = median(diabetes[[col_name]], na.rm = TRUE),
      SD       = sd(diabetes[[col_name]], na.rm = TRUE),
      Variance = var(diabetes[[col_name]], na.rm = TRUE)
    )
    for (stat_name in names(stats_vector)) {
      cat(sprintf("%-8s: %0.2f\n", stat_name, stats_vector[[stat_name]]))
    }
  } else {
    cat("  (non‑numeric column, skipping)\n")
  }
  cat("\n")
}
